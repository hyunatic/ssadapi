Link Example:
https://ssadapi.hyunatic.com/public/index.php/api/login

All link will return a JSON array

Authentication:
// This is email login //
/api/login [POST]
{
    "email" : string,
    "password" : string
}

// This is Facebook login //
/api/fblogin [POST]
{
    "fbid": string
}

// User Registration //
/api/register [POST]
{
    "fbid": string,
    "email": string,
    "password": string,
    "name": string,
    "usertype": string,
    "tutgrp": string
}

// Get Student Information for Marking and Profile Page //
/api/studentinfo [POST]
{
    "id" : string
}

// Get Professor Information for Profile Page //
/api/profinfo [POST]
{
    "id" : string
}

// Update Profile for Professor or Student //
/api/profile/update [POST]
{
    "id": string,
    "name": string,
    "fbid": string,
    "email": string,
    "picture": string,
    "tutgrp": stirng
}

// Link user account with facebook account //
/api/facebook/link [POST]
{
    "id" : string,
    "fbid" : string
}

LeaderBoard:
// Get Leader Scores //
/api/leaderboard [GET]

// Get all Student doing the current Tutorial //
/api/tut/student [POST]
{
    "tutid" : string
}

// Get all the information for Professor to mark in the Marking page //
/api/student/submission [POST]
{
    "id" : string
}

// Unity Game application to submit student's answers //
/api/submission [POST]
{
    "name" : string,
    "image" : string,
    "date" : string,
    "tutgrp" : string,
    "tutid" : string,
    "studid" : string,
}

// Professor to mark student's score and give comments //
/api/marking/update [POST]
{
    "id" : string,
    "score" : string,
    "comment" : string
}
